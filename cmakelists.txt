cmake_minimum_required(VERSION 3.10)

project(kale)

if(MSVC)
  add_compile_options(/W4 /WX)
else()
  add_compile_options(-Wall -Wextra -Wpedantic -Werror)
endif()

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED True)

file(GLOB_RECURSE KALE_SOURCES "kale/*.c" "kale/*.h")
file(GLOB_RECURSE META_SOURCES "meta/*.c" "meta/.*.h")
file(GLOB_RECURSE FRONTEND_SOURCES "frontend/*.c" "frontend/*.h")

add_executable(meta ${META_SOURCES})
add_library(kale STATIC ${KALE_SOURCES})
add_executable(frontend ${FRONTEND_SOURCES})

add_custom_command(
  TARGET meta POST_BUILD
  COMMAND meta
  WORKING_DIRECTORY ${CMAKE_CURRENT_LIST_DIR}
  COMMENT "Running meta program"
)

add_dependencies(kale meta)

target_link_libraries(frontend PRIVATE kale)

target_include_directories(meta PRIVATE "meta")
target_include_directories(frontend PRIVATE "kale")
